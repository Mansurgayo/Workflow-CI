name: MLflow CI Advanced

on:
  push:
    branches: ["main"]
  workflow_dispatch:

jobs:
  train_and_deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v3

      - name: Set up Python 3.12
        uses: actions/setup-python@v4
        with:
          python-version: "3.12.7"

      - name: Check Env
        run: |
          python --version
          echo "Directory:"
          ls -R

      - name: Install dependencies
        run: |
          pip install mlflow scikit-learn pandas numpy

      - name: Run mlflow project
        working-directory: MLProject
        run: mlflow run .

      - name: Get latest MLflow run_id
        id: getrun
        run: |
          RUN_ID=$(ls -t mlruns/0 | head -n 1)
          echo "run_id=$RUN_ID" >> $GITHUB_OUTPUT

      # ----------------------------------------------------
      # UPLOAD ARTIFACT TO GOOGLE DRIVE (Rclone Method)
      # ----------------------------------------------------
      - name: Install rclone
        run: |
          sudo apt install rclone -y

      - name: Configure rclone
        run: |
          echo "${{ secrets.RCLONE_CONF }}" > rclone.conf

      - name: Upload to Google Drive
        run: |
          rclone --config=rclone.conf copy MLProject/mlruns/0/${{ steps.getrun.outputs.run_id }} gdrive:mlflow_artifacts/${{ steps.getrun.outputs.run_id }}
      # ----------------------------------------------------

      - name: Build Docker Model
        working-directory: MLProject
        run: |
          mlflow models build-docker -m mlruns/0/${{ steps.getrun.outputs.run_id }}/artifacts/model -n mlflow_model_image

      # ----------------------------------------------------
      # Docker Hub Login + Push
      # ----------------------------------------------------
      - name: Log in to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      - name: Tag Docker Image
        run: |
          docker tag mlflow_model_image ${{ secrets.DOCKERHUB_USERNAME }}/mlflow_model_image:latest

      - name: Push Docker Image
        run: |
          docker push ${{ secrets.DOCKERHUB_USERNAME }}/mlflow_model_image:latest

      # ----------------------------------------------------
      # Closing steps
      # ----------------------------------------------------
      - name: Post Log in to Docker Hub
        run: echo "Docker push completed."

      - name: Complete job
        run: echo "CI Workflow finished."
